    if (element?.article && typeof element.article === 'string') {
      element.article = element.article.includes('; ')
        ? element.article.split('; ')
        : [element.article];
    }

    if (element?.objectID) {
      if (!element.article || !Array.isArray(element.article)) {
        element.article = [];
      }

      if (!element.article.includes(element.objectID)) {
        element.article.push(element.objectID);
      }

      delete element.objectID;
    }

    if (element?.infos && Array.isArray(element.infos)) {
      element.infos = element.infos
        .map((info: any) => {
          if (Array.isArray(info.items)) {
            info.items = info.items
              .map((item: any) => {
                delete item.type;

                for (const key of Object.keys(item)) {
                  if (
                    item[key] === '' ||
                    item[key] === null ||
                    item[key] === undefined
                  ) {
                    delete item[key];
                  }
                }

                if (info?.code === 'description' && item.value === element?.description) {
                  delete item.title;
                  delete item.code;
                  delete item.value;
                } else if (info?.code === 'product_details') {
                  if (item?.title === 'PID' && this.isEmpty(item?.value)) {
                    delete item.title;
                  }
                  if (item?.title === 'Article') {
                    if (!Array.isArray(element.article)) {
                      element.article = this.isEmpty(element.article)
                        ? []
                        : [String(element.article)];
                    }

                    if (!this.isEmpty(item.value) && !element.article.includes(item.value)) {
                      element.article.push(item.value);
                    }

                    if (!this.isEmpty(item.value) && element.article.includes(item.value)) {
                      delete item.title;
                      delete item.value;
                    }
                  }
                  if (item?.title === 'EAN' && element?.gtin === item?.value) {
                    delete item.title;
                    delete item.value;
                  }
                }
                return item;
              })
              .filter((item: any) => Object.keys(item).length > 0);

            if (info.items.length === 0) delete info.items;
          }
          else if (info?.items && typeof info.items === 'object') {
            delete info.items.type;

            for (const key of Object.keys(info.items)) {
              if (
                info.items[key] === '' ||
                info.items[key] === null ||
                info.items[key] === undefined
              ) {
                delete info.items[key];
              }
            }

            if (Object.keys(info.items).length === 0) delete info.items;
          }

          if (!info.items) {
            delete info.code;
            delete info.title;
          }

          return info;
        })
        .filter((info: any) => Object.keys(info).length > 0);

      if (element.infos.length === 0) delete element.infos;
    }
